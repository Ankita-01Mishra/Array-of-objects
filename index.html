Question 1 : We are interested in retrieving only the name of the students and all the names should be in caps. ['JOHN', 'BABA', 'YAGA', 'WICK'] 

    Method-1

    let names = [];
    studentRecords.forEach(element =>{
        element=element.name;
        names.push(element.toUpperCase());
    }) 
    console.log(names);
   

    Method-2

    namesCap = studentRecords.map( studentName => studentName.name.toUpperCase());
    console.log(namesCap);
